# Write your own code

bool_cond = True

currencies["bool_cond"] = (currencies["growth_euro"] > 2) | (currencies["growth_euro"] < -2)

plt.plot(currencies["date"], currencies["growth_euro"], label="Growth Euro")
plt.fill_between(currencies["date"], currencies["growth_euro"], where=currencies["bool_cond"], color="red", alpha=0.3, label="Extreme Growth")
